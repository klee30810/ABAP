FORM get_airline_data .

** Get airline data
  CLEAR : gv_carrname, gv_currcode, gv_url.
  SELECT SINGLE carrname currcode url
    INTO ( gv_carrname, gv_currcode, gv_url )        "### ##### ##### corresponding ##
    FROM scarr
    WHERE carrid EQ gv_carrid.

** Set icon
  CASE gv_carrid.
    WHEN 'AA' OR 'DL'.
      gv_icon = icon_ws_plane.
    WHEN 'KA'.
      gv_icon = icon_activate.
      gv_url = 'hello world!'.
    WHEN OTHERS.
      CLEAR gv_icon.
  ENDCASE.


ENDFORM.
