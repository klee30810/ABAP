MODULE modify_screen OUTPUT.

  LOOP AT SCREEN.

** Intensified
    IF screen-group1 EQ 'GR1'.
      screen-intensified = 1.       " Text color is intensified (blue).
      MODIFY SCREEN.
    ENDIF.

** Display & Edit
    IF screen-group1 EQ 'RD1'.
      CASE 'X'.
        WHEN gv_disp. " Display mode
          screen-input = 0.

        WHEN gv_edit. " Edit mode
          screen-input = 1.

      ENDCASE.

      MODIFY SCREEN.

    ENDIF.

** Show & Hide
    IF screen-group2 EQ 'RD2'.
      CASE 'X'.
        WHEN gv_show. " Show mode
          screen-active = 1.

        WHEN gv_hide. " Hide mode
          screen-active = 0.

      ENDCASE.

      MODIFY SCREEN.

    ENDIF.

  ENDLOOP.




ENDMODULE.
