FUNCTION zfm_bseg_a12_02.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     VALUE(IV_BUKRS) TYPE  BUKRS
*"     VALUE(IV_GJAHR) TYPE  GJAHR
*"     VALUE(IV_BELNR) TYPE  BELNR_D
*"  EXPORTING
*"     VALUE(EV_SUBRC) TYPE  CHAR01
*"  TABLES
*"      ET_ITEM STRUCTURE  ZBSEGA12_02
*"  EXCEPTIONS
*"      EMPTY_PARAMETER
*"----------------------------------------------------------------------

*       Global data declarations
  CASE space.
    WHEN iv_bukrs OR iv_gjahr OR iv_belnr.
      RAISE empty_parameter.
  ENDCASE.

  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE et_item
    FROM bseg
    WHERE bukrs EQ iv_bukrs
      AND gjahr EQ iv_gjahr
      AND belnr EQ iv_belnr.

  IF sy-subrc EQ 0.
    ev_subrc = 'S'.
  ELSE.
    ev_subrc = 'F'.

  ENDIF.

ENDFUNCTION.
